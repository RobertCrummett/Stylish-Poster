\documentclass{cgem-poster}
\graphicspath{{Logo/}}

%----------------------------------
% Title section
\title{An informative title that is also catchy}
\author{Author One \and Author Two}
\date{\today}

%----------------------------------
% Set the poster text box alignment/geometry
\lua{set}{ background margin = 0.75 }
\lua{set}{ logo margin = 0.75 }
\lua{set}{ title height = 8 }

%----------------------------------
% Build the main poster body

\poster {
  % Set up the upper title box
  \postertitle \inserttitle \insertauthor \insertdate

  % Three major columns
  \newcommand{\colwidth}{%
    \dimexpr(\lua{get}{internal margin width}-
      (\lua{get}{internal margin value}*2))/3\relax%
  }

  % Column one background box
  \begin{textblock*}{\colwidth}
    ( \lua{get}{internal margin left}, \lua{get}{internal margin top} )
    \textblockcolor{black}
    \begin{minipage}[t][ \lua{get}{ internal margin height } ][t]{\colwidth} \end{minipage}
  \end{textblock*}

  % Introduction -- column one
  \begin{textblock*}{\dimexpr\colwidth-2cm\relax}
    (\dimexpr\lua{get}{internal margin left}+1cm\relax,
      \dimexpr\lua{get}{internal margin top}+1cm\relax) 
    \textblockcolor{black}
    \begin{minipage}[t]{\dimexpr\colwidth-2cm\relax}
      \begin{center}
        \textbf{\large Introduction}
      \end{center}
      This is my first paragraph.
      This is my first paragraph.
      This is my first paragraph.
      This is my first paragraph.
      This is my first paragraph.

    \end{minipage}
  \end{textblock*}

  % Column two background box
  \begin{textblock*}{\colwidth}
    (\dimexpr \lua{get}{internal margin left} + \colwidth + \lua{get}{internal margin value} \relax, \lua{get}{internal margin top})
    \textblockcolor{black}
    \begin{minipage}[t][\lua{get}{internal margin height}][t]{\colwidth}\end{minipage}
  \end{textblock*}

  \begin{textblock*}{\dimexpr\colwidth-2cm\relax}
    (\dimexpr\lua{get}{internal margin left}+\colwidth+\lua{get}{internal margin value}+1cm\relax,
    \dimexpr\lua{get}{internal margin top}+1cm\relax)
    \textblockcolor{black}
    \begin{minipage}{\dimexpr\colwidth-2cm\relax}
      \begin{center}
        \textbf{\large Methodology}
      \end{center}
      Hello world.
      Hello world.
      Hello world.
      Hello world.
      Hello world.
      Hello world.
      Hello world.
      Hello world.
      Hello world.

    \end{minipage}
  \end{textblock*}


  % Column three background box
  \begin{textblock*}{\colwidth}
    (\dimexpr \lua{get}{internal margin left} + (\colwidth * 2) + (\lua{get}{internal margin value} * 2)\relax, \lua{get}{internal margin top})
    \textblockcolor{black}
    \begin{minipage}[t][\lua{get}{internal margin height}][t]{\colwidth}\end{minipage}
  \end{textblock*}
 
} % poster
