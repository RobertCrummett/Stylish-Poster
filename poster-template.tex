% Example of a CGEM poster
\documentclass{cgem-poster}
\usepackage[base]{babel}
\usepackage{lipsum}

%----------------------------------
% Title section
\title{An informative title that is also catchy}
\author{Author One \and Author Two}
\date{\today}

%----------------------------------
% Set the poster text box alignment/geometry
\lua{set}{ background margin = 1 }

%----------------------------------
% Build the main poster body

\poster {
  % Set up the upper title box
  \postertitle \inserttitle \insertauthor \insertdate

  %% Column two
  \begin{textblock*}{0.3\paperwidth}(0.345\paperwidth,0.115\paperheight)
    \textblockcolor{pink}
    \begin{minipage}[t][0.8\paperheight][t]{0.3\pagewidth}
      \begin{center}
        Hello world
      \end{center}
    \end{minipage}
  \end{textblock*}
  
  %% Column three
  \begin{textblock*}{0.3\paperwidth}(0.665\paperwidth,0.115\paperheight)
    \textblockcolor{black}~\vspace{0.845\paperheight}
  \end{textblock*}

} % poster
